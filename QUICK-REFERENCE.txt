━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

THE PROBLEM:
  • 11-12 files sourced unconditionally (lines 32-44 of plan-db.sh)
  • Total: 3,111 lines of code loaded per invocation
  • Even `plan-db list` loads plan-db-import.sh (425 lines, not needed)
  • Startup overhead: 100+ milliseconds

FILES SOURCED:
  plan-db-core.sh (246), plan-db-crud.sh (sources 4 modules), 
  plan-db-validate.sh (153), plan-db-display.sh (287),
  plan-db-import.sh (425), plan-db-drift.sh (152),
  plan-db-conflicts.sh (199), plan-db-cluster.sh (219),
  plan-db-remote.sh (174), plan-db-delegate.sh, 
  plan-db-intelligence.sh (271), plan-db-knowledge.sh (176)

THE SOLUTION:
  Move sources into case statement by subcommand
  
  Before:
    source lib/plan-db-core.sh
    source lib/plan-db-crud.sh
    ... 11 sources ...
  
  After:
    case "${1:-}" in
      list|create) source lib/plan-db-core.sh; source lib/plan-db-crud.sh ;;
      validate*) source lib/plan-db-core.sh; source lib/plan-db-validate.sh ;;
      import) source lib/plan-db-core.sh; source lib/plan-db-crud.sh; source lib/plan-db-import.sh ;;
    esac

IMPACT: 100ms → 15-30ms startup (-70%)
EFFORT: 30 minutes
FILES:  scripts/plan-db.sh only

2. SUBPROCESS EXPLOSION (8-30+ processes per operation)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

THE PROBLEM:
  • plan-db-import.sh: 51 individual sqlite3 calls
  • plan-db-update.sh: 40 individual sqlite3 calls
  • Each call = fork + parse + execute + exit overhead
  • No transaction batching

EXAMPLE (plan-db-import.sh):
  sqlite3 "$DB_FILE" "INSERT INTO tasks..."      # Fork 1
  sqlite3 "$DB_FILE" "INSERT INTO tasks..."      # Fork 2
  sqlite3 "$DB_FILE" "INSERT INTO tasks..."      # Fork 3
  ... 48 more individual calls ...

THE SOLUTION:
  Use SQLite transaction batching
  
  Before: 51 individual sqlite3 calls
  
  After:
    sqlite3 "$DB_FILE" << SQL
    BEGIN TRANSACTION;
    INSERT INTO tasks...
    INSERT INTO tasks...
    ... (all 50 inserts)
    SELECT COUNT(*) ...
    COMMIT;
    SQL
  
  Result: 51 calls → 2 calls (fork once, commit once)

IMPACT: 30 processes → 5-10 processes (-80%)
        3-4× faster bulk operations
EFFORT: 1-2 hours
FILES:  plan-db-query.sh (new), plan-db-import.sh, plan-db-update.sh

3. HOOK OVERHEAD (100ms latency)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

THE PROBLEM:
  • 37 hooks total
  • 8-10 hooks run per Copilot tool call
  • 20+ hooks × 5ms average = 100ms added latency

HEAVY HOOKS:
  secret-scanner.sh (208 lines, 20+ regex patterns)     → 15ms
  verify-before-claim.sh (146 lines, 2-3 DB queries)   → 10ms
  enforce-standards.sh (104 lines, 10+ regex checks)    → 5ms
  session-tokens.sh (82 lines, SQLite INSERT)           → 5ms

THE SOLUTION:
  1. Cache secret-scanner regex compilation
     Create: /tmp/secret-scanner-compiled.regex
     TTL: 1 hour
     Saves: 10ms per execution

  2. Batch token tracking (async, not per-session)
     Move: token recording to background job
     Saves: 5ms per session end

  3. Optimize verify-before-claim DB queries
     Add: Check if already verified (cached)
     Saves: 3ms per claim

IMPACT: 100ms latency → 5-10ms (-95%)
EFFORT: 2 hours
FILES:  hooks/secret-scanner.sh, hooks/session-tokens.sh, hooks/verify-before-claim.sh

4. DIGEST DUPLICATION (16 identical scripts)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

THE PROBLEM:
  16 separate scripts with identical caching pattern:
  
  db-digest.sh          ci-digest.sh           service-digest.sh
  build-digest.sh       audit-digest.sh        sentry-digest.sh
  test-digest.sh        npm-digest.sh          migration-digest.sh
  git-digest.sh         deployment-digest.sh   copilot-review-digest.sh
  pr-digest.sh          error-digest.sh        merge-digest.sh
  
  All share:
    source lib/digest-cache.sh
    CACHE_TTL=<varying>
    digest_cache_get()
    <run tool>
    digest_cache_set()

THE SOLUTION:
  Create single dispatcher: scripts/digest.sh
  
  Usage:
    digest.sh db <args>       # calls db-digest handler
    digest.sh build <args>    # calls build-digest handler
    digest.sh test <args>     # calls test-digest handler
    digest.sh git <args>      # calls git-digest handler
    ... etc ...
  
  Single digest-cache.sh implementation, modular handlers

IMPACT: 16 scripts → 1 dispatcher
        Delete 15 files (~500 lines)
EFFORT: 1 hour
FILES:  scripts/digest.sh (new), delete 15 original *-digest.sh scripts

6. DUPLICATE QUERY LOGIC (~150+ inline SQL calls)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

THE PROBLEM:
  Query distribution across modules:
    plan-db-import.sh:   51 sqlite3 calls
    plan-db-update.sh:   40 sqlite3 calls
    plan-db-create.sh:   14 sqlite3 calls
    plan-db-display.sh:   5 sqlite3 calls
    plan-db-read.sh:      3 sqlite3 calls
    ────────────────────────────────
    Total:              ~150 inline SQL calls
  
  Issues:
    1. No centralized query builder
    2. sql_escape() called inconsistently (SQL injection risk)
    3. Same queries written multiple times (DRY violation)

THE SOLUTION:
  Create plan-db-query.sh helper module
  
  Implement:
    sql_insert_batch()
    sql_update_batch()
    sql_select()
    sql_delete()
  
  Benefits:
    1. Consistent escaping (single point of failure prevention)
    2. Single implementation (easier to maintain)
    3. Transaction support built-in
    4. Less error-prone

IMPACT: Reduce duplication 66%
        Improve security (centralized escaping)
EFFORT: 2 hours
FILES:  plan-db-query.sh (new), plan-db-import.sh, plan-db-update.sh

7. WORKTREE LOOKUPS (2-3 queries per operation)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

THE PROBLEM:
  Current pattern for each task update:
    sqlite3 "$DB_FILE" "SELECT worktree_path FROM plans WHERE id=$plan_id"
    sqlite3 "$DB_FILE" "SELECT worktree_path FROM waves WHERE id=$wave_id"
    sqlite3 "$DB_FILE" "SELECT worktree_path FROM tasks WHERE id=$task_id"
  
  Same worktree_path fetched multiple times
  (Would hit main worktree, wave worktree, task worktree in that order)

THE SOLUTION:
  Cache worktree paths in /tmp/plan-db-worktrees-$(date +%s).cache
  
  Logic:
    1. Check /tmp/plan-db-worktrees-*.cache (TTL: 5 minutes)
    2. If miss, query DB (maintain triple hierarchy)
    3. Store in cache with TTL
    4. Return from cache next time

IMPACT: 2-3 queries → 1 query + cache hit (50% fewer ops)
EFFORT: 30 minutes
FILES:  scripts/lib/plan-db-core.sh (add caching layer)
